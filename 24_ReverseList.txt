/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode(int x) : val(x), next(NULL) {}
 * };
 */
class Solution {
public:
    ListNode* reverseList(ListNode* head) {
        if(head == NULL || head->next ==NULL)
            return head;
        ListNode* cur = nullptr;
        ListNode* q = head;
        //依次断链再接上
        while(q!=nullptr){
            ListNode* t = q->next;
            q->next = cur;
            cur = q;
            q = t;
        }
        return cur;
    }
};

class Solution {
public:
    ListNode* reverseList(ListNode* head) {
        if(head == NULL || head->next ==NULL)
            return head;
        ListNode* p = head;
        ListNode* q = head;
        ListNode* temp = head;
        while(p->next!=NULL)
             p = p->next;
        while(q!=p){
            temp = q;
            q = q->next;
            temp->next = p->next;
            p->next = temp;
        }
        return p;
    }
};
