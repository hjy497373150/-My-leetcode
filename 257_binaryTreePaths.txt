/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    vector<string> binaryTreePaths(TreeNode* root) {
        string path = "";
        vector<string> paths;
        BFS(root,path,paths);
        return paths;
    }
    void BFS(TreeNode* root,string path,vector<string>& paths){
        if(root!=nullptr){
            path += to_string(root->val);
            if(root->left==nullptr && root->right==nullptr)
                paths.push_back(path);
            else{
                path += "->";
                BFS(root->left,path,paths);
                BFS(root->right,path,paths);
            }
        }
    }
};
