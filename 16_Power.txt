class Solution {
public:
    double myPow(double x, int n) {
        //¿ìËÙÃİ
        if(n==0)
            return 1;
        long N = n;
        if(n<0){
            N = -N;
            x = 1/x;
        }
        double res = 1;
        while(N>0){
            if((N&1)==1){
                res = res * x;
            }
            x *= x;
            N = N>>1;
        }
        return res;
    }
};
