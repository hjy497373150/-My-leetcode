class Solution {
public:
    int minArray(vector<int>& numbers) {
        if(numbers.size()==1)
            return numbers[0];
        int left = 0;
        int right = numbers.size()-1;
        while(left<right){
            int mid = left +(right-left)/2;
            if(numbers[mid]>numbers[right])
                left = mid+1;
            else if(numbers[mid]==numbers[right])
                right = right-1;
            else
                right = mid;
        }
        return numbers[left];
    }
};
